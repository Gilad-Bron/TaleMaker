Entity 2 of 8 — Character

A player's character. Can be global or tied to a specific tale.

Identity
{
  id:         string
  userId:     string
  taleId?:    string    // null = global
  avatarUrl?: string
  name:       string
  bio:        string
  level:      number
  experience: number
  isAlive:    boolean
  gold:       number
  createdAt:  Date
}
Resources
{
  health: {
    current: number
    max:     number
  }
  mana: {
    current: number
    max:     number
  }
}
Ability Scores
{
  abilityScores: {
    STR: number
    INT: number
    CHA: number
    DEX: number
    WIS: number
    CON: number
  }
}
Saving Throws
{
  savingThrows: {
    fortitude: number  // STR or CON
    reflex:    number  // DEX or INT
    will:      number  // WIS or CHA
  }
}
Skills
{
  skills: {
    athletics:     number  // STR
    acrobatics:    number  // DEX
    stealth:       number  // DEX
    persuasion:    number  // CHA
    deception:     number  // CHA
    intimidation:  number  // CHA
    insight:       number  // WIS
    perception:    number  // WIS
    investigation: number  // INT
    arcana:        number  // INT
    lore:          number  // INT
  }
}
Feats
{
  feats: [
    {
      id:          string
      name:        string
      description: string
      effect: {
        type:  string   // e.g. "hp_bonus"
        value: number   // e.g. 10
      }
    }
  ]
  inventory: InventoryItem[]
}
