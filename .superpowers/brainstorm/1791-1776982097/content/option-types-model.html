Option Type Data Model

Each option in a dialog interaction has a required type field

Base Option Shape
{
  type:  "speak" | "action" | "skill_check" | "combat" | "rest" | "shop" | "loot" | "examine"
  label: string   // quoted = speech, plain = italic action
  next?: string   // dialog node key to navigate to
}
Per-Type Extensions
Speak no extra fields 
{ type: "speak", label: '"Who are you?"', next: "ask" }
Action no extra fields 
{ type: "action", label: "Step forward slowly", next: "forward" }
Skill Check 
{
  type: "skill_check",
  label: "Slip past...",
  skill: { stat: "AGI", dc: 12 },
  successNext: "slipped",
  failText:    "string"
}
Combat 
{
  type:  "combat",
  label: "Draw a weapon",
  enemy: "stranger"
}
Shop 
{
  type:   "shop",
  label:  "Browse wares",
  shopId: "merchant_01"
}
Loot 
{
  type:   "loot",
  label:  "Search the body",
  lootId: "body_01"
}
Rest 
{
  type:  "rest",
  label: "Camp for the night",
  next:  "morning"
}
Examine Surroundings 
{
  type:   "examine",
  label:  "Examine the gate",
  areaId: "gate_ruins"
}
