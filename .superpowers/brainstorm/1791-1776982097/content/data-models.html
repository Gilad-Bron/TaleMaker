Data Models

Full platform schema — Tale → Act → Chapter → Interaction → Option

User
{
  id:       string
  email:    string
  name:     string
  roles:    "author" | "player" | "both"
  createdAt: Date
}
Character
{
  id:        string
  userId:    string        // owner
  name:      string
  backstory: string
  stats: {
    STR INT CHA AGI WIS CON: number
  }
  hp:        number
  maxHp:     number
  mana:      number
  maxMana:   number
  gold:      number
  inventory: InventoryItem[]
}
InventoryItem
{
  id:       string
  name:     string
  type:     "weapon" | "armor" | "consumable" | "key" | "misc"
  quantity: number
}
Game Session
{
  id:                   string
  userId:               string
  taleId:               string
  characterId:          string
  currentInteractionId: string
  failedOptions:        string[]  // option refs
  flags:                Record<string, boolean>
  savedAt:              Date
}
Tale
{
  id:           string
  authorId:     string
  title:        string
  description:  string
  visibility:   "public" | "private" | "invite"
  inviteList:   string[]           // userIds
  characterMode: "new" | "premade" | "global_premade" | "existing"
  premadeCharacters: Character[]   // if premade
  context: {                         // used for AI prompt generation
    genre:       string
    tone:        string
    setting:     string
    worldDetails: string
    themes:      string[]
  }
  acts: Act[]
  createdAt: Date
}
Act
{
  id:       string
  taleId:   string
  title:    string
  order:    number
  chapters: Chapter[]
}
Chapter
{
  id:           string
  actId:        string
  title:        string
  order:        number
  interactions: Interaction[]
}
Interaction
{
  id:        string
  chapterId: string
  title:     string
  location:  string
  text:      string              // narrator prose
  prompt: {                      // AI generation config
    mode: "free_text" | "mechanics" | "context"
    value: string               // free text, or serialised selections
  }
  options: Option[]
}
Option
{
  type:  "speak" | "action" | "skill_check" | "combat" | "rest" | "shop" | "loot" | "examine"
  label: string                   // quoted = speech, plain = action (italic)
  next?: string                   // interaction id to navigate to

  // skill_check only
  skill?:       { stat: "STR"|"INT"|"CHA"|"AGI"|"WIS"|"CON", dc: number }
  successNext?: string
  failText?:    string

  // combat only
  enemy?:   string               // enemy ref id

  // shop / loot / examine only
  refId?:   string               // shopId | lootId | areaId
}
